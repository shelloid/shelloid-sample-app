var assert = require("assert");
modules.export = 
{

	"req.body":
	{
	   "userid" : "integer",
	   "comment": ["string", "opt"]
	},
	
	"req.query" : {"name": "string"},
	
	"req.validate" : function(req){
		assert(req.body.userid <= 10);
	}

	"resp.body":
	{
		"friends" : 
			[{"friendId": "integer"; "friendName": "string"}, "opt"]
	}
	
	"resp.validate" : function(resp){
		assert(resp.body.friends.length <= 100);
	}
}
